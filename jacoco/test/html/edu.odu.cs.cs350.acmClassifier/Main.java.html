package edu.odu.cs.cs350.acmClassifier;

/**
 * The class whose sole purpose is to house the main function
 * @author Team 7
 *
 */
public class Main
{
	//take in files
	//tell me if files are valid
	//generate a document
	//parse the document
	
	/**
	 * The main function, what runs when you run the program
	 * @param args the names of the files we use to create documents
	 */
	public static void main(String[] args)
	{
		
		//first, tell us if everything is valid
		
		boolean valid = true;    //whether all files are valid
		for(String arg : args) { //loop through all file names
			if(!DocumentReader.isValid(arg)) { //if not valid
				System.out.println(arg+" is not valid"); valid=false; //print message, set valid to false
			}
		}
		if(valid) { System.out.println("All files are valid"); } //if all files were valid, print message
		System.out.println();                                    //add another empty line
		
		
		
		//now, parse the documents
		
		Document[] docs = new Document[args.length]; //create array of documents
		for(int n=0;n<args.length;n++) {             //loop through all files
			docs[n] = DocumentReader.readDocument(args[n]); //set each document
		}
		
		StopList list = new StopList(); //create stop list
		
		for(Document doc : docs) { //loop through all documents
			Parser.toLowerCase(doc);         //make raw text lower case
			Parser.removePunctuation(doc);   //remove punctuation from raw text
			Parser.pruneStopList(doc, list); //remove all stop words
		}
		
		System.out.println("Successful!"); //tell us everything was successful
	}
}
