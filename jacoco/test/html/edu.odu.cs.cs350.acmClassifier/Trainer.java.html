package edu.odu.cs.cs350.acmClassifier;

/**
 * Class used to retrain/ fine tune ML network
 * @author Team 7
 */

public class Trainer
{
  private Document trainingDoc; // document used for training

  /**
   * Default Constructor
   */
  public Trainer()
  {
    trainingDoc = new Document();
  }

  /**
   * Constructor given training document
   * 
   * @param doc Document to be used for training
   */
  public Trainer(Document doc)
  {
    trainingDoc = doc;
  }

  /**
   * Returns training document
   * 
   * @return the training document
   */
  public Document getTrainingDoc()
  {
    return trainingDoc;
  }

  /**
   * Sets training document
   * 
   * @param doc the document to be used for training
   */
  public void setTrainingDoc(Document doc)
  {
    trainingDoc = doc;
  }

  /**
   * Returns hashCode
   * 
   * @return the hashCose of each private member
   */
  public int hashCode()
  {
    return trainingDoc.hashCode();
  }

  /**
   * Trainer clone function
   * 
   * @return a copy of Trainer
   */
  public Trainer clone()
  {
    return new Trainer(trainingDoc);
  }

  /**
   * Check if an object equals trainer
   * 
   * @return true if objects equal/ false if not
   */
  public boolean equals(Object other)
  {
    if(!(other instanceof Trainer)) // if not a trainer
    {
      return false;
    }

    Trainer rhs = (Trainer) other;

    return this.trainingDoc.equals(rhs.trainingDoc);
  }

  /**
     * Turns trainer into a string
     * 
     * @return the buffer as a string
     */
  public String toString()
  {
    StringBuffer buff = new StringBuffer("{");
    buff.append(trainingDoc);
    buff.append("}");

    return buff.toString();
  }

  /*
  Is a class for the trainers for our neural networks. Given a set of documents, as well as the correct classifications for each document, it's going to
  ask each of a set of neural networks to determine the classification for each document, then grade them based on how well they did (the grade is known
  as the "fitness function"). Each generation, it'll grade a set of better and better neural networks, thus it can be expected that they will have higher
  and higher fitness scores. We can stop once the neural networks achieve an accuracy of about 80%.
  */
}
