package edu.odu.cs.cs350.pne;

import java.io.File;
import java.util.Scanner;
import java.util.HashSet;

/**
 * List containing all the stop words (common words)
 * There will only be one instance of stop list, for our purposes at least
 * 
 * @author Chris, Kayla
 */
public class StopList
{
	/***
	 * Stores all the unneeded words
	 */
	private HashSet<String> stopList = new HashSet<String>();
	
	/***
	 * Default constructor, cenerates a stop list from a file
	 * 
	 * @param fileName file containing stop words
	 * 
	 * @throws RuntimeException if file does not exists or cannot be opened
	 */
	public StopList(String fileName)
	{ 
		Scanner scan; 
		try { scan = new Scanner(new File(fileName)); } //try to initialize scanner
		catch(Exception ex) { throw new RuntimeException("Cannot open file \""+fileName+"\""); }
		
		// while file file has next line
		while(scan.hasNextLine())
		{ 
			String line = scan.nextLine();
			
			String[] split = line.split(" ");                  //split line into chunks separated by spaces
			for(String s : split) { stopList.add(s.toLowerCase()); } //add each word to our list
		}
		
		scan.close();
		
		stopList.add(""); //add the empty string to the stop list to apply a special case pruning measure
	}

	/***
	 * Constructor, sets existing stop list to current stop list
	 * 
	 * @param stop existing stop list
	 */
	public StopList(HashSet<String> stop)
	{
		for(String s : stop) { stopList.add(s.toLowerCase()); }
	}


	/***
	 * Retreve stop list
	 * 
	 * @return the stop list
	 */
	public HashSet<String> getStoplist()
	{
		return stopList;
	}

	/***
	 * Set stop list to existing stop list
	 *  
	 * @param sl new stop list
	 */
	public void setStoplist(HashSet<String> sl)
	{
		stopList.clear();

		for(String s : sl) { stopList.add(s.toLowerCase()); }
	}


	/***
	 * Adds a word to the stop list
	 * 
	 * @param word word to be added to the stop list
	 */
	public void addToStopList(String word)
	{
		stopList.add(word.toLowerCase());
	}

	/***
	 * Removes a word from the stop list
	 * 
	 * @param word word to be removed from the stop list
	 */
	public void removeFromStopList(String word)
	{
		stopList.remove(word.toLowerCase());
	}
	

	/***
	 * Returns stop list in string format
	 */
	@Override
	public String toString()
	{
		String result = "";
		for(String s : stopList) { result+=s+", "; }
		return result;
	}
	
	/***
	 * Check if stop list contains words
	 * 
	 * @param word word to search for in list
	 * @return true if found, false if not found
	 */
	public boolean contains(String word)
	{
		return stopList.contains(word.toLowerCase()); //return true iff stoplist contains word
	}
}
