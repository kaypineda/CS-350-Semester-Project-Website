package edu.odu.cs.cs350.pne;

import java.util.ArrayList;

/**
 * Document class that has the file name, the list of words needed for classification
 * and the list of classifications
 * 
 */
public class Document implements Iterable<String>
{
    private String name;
    private ArrayList<String> words;
    private ArrayList<Character> classes; //array of classifications
    
    /**
     * Default constructor 
     */
    public Document() 
    {
        name = "file.txt";
        words = new ArrayList<String>();
        classes = new ArrayList<Character>();
    }
    
    /**
     * Constructor with classification added 
     * 
     * @param n name of the document
     */
    public Document(String n) 
    {
        name = n;
        words = new ArrayList<String>();
        classes = new ArrayList<Character>();
    }

    /**
     * Constructor to be used for AI training
     * 
     * @param n name of the document
     * @param c letter for classification purposes 
     */
    public Document(String n, ArrayList<Character> c) 
    {
        name = n;
        words = new ArrayList<String>();
        classes = c;
    }

    /**
     * Constructor for Document
     * 
     * @param n name of the document
     * @param w contains completed parsed words
     * @param c contains classification character
     */
    public Document(String n, ArrayList<String> w, ArrayList<Character> c)
    {
        name = n;
        words = w;
        classes = c;
    }

    /**
     * Gets the name of the document
     * 
     * @return returns name of the document
     */
    public String getName()
    {
        return name;
    }

    /**
     * Gets the parsed arraylist of words
     * 
     * @return returns the parsed words
     */
    public ArrayList<String> getWords()
    {
        return words;
    }

    /**
     * Returns the classification list
     * 
     * @return classes  
     */
    public ArrayList<Character> getClasses()
    {
        return classes;
    }

    /**
     * Sets the document name
     * 
     * @param n document name
     */
    public void setName(String n)
    {
        name = n;
    }

    /**
     * Stores the list of parsed words
     * 
     * @param w arraylist of parsed words
     */
    public void setWords(ArrayList<String> w)
    {
        if(words!=null) { words = w; }
    }

    /**
     * Sets the clssification letter of the document
     * 
     * @param c arraylist of the character classification 
     */
    public void setClasses(ArrayList<Character> c)
    {
        classes = c;
    }

    /**
     * Adds the letter for the classification
     * 
     * @param c the letter for classification 
     */
    public void addClass(Character c)
    {
        classes.add(c);
    }

    /**
     * Removes the letter for the classification
     * 
     * @param c class to be removed
     */
    public void removeClass(Character c)
    {
        for (int i = 0; i < classes.size(); i++)
        {
            if (classes.get(i).equals(c))
                {
                    classes.remove(i);
                    return;
                }
        }
    }

    /**
     * Checks if hashCode is equal
     * 
     * @return the hashCode of each private member
     */
    public int hashCode()
    {
        return name.hashCode() + words.hashCode() + classes.hashCode();
    }

    /**
     * Document clone function
     * 
     * @return a copy of Document
     */
    public Document clone()
    {
        return new Document(name, (ArrayList)words.clone(), (ArrayList)classes.clone()); //this ASSUMES that neither arraylists should be null
    }

    /**
     * Making sure document will be equal to another document object
     * 
     * @return the set rules of what will make them equal
     */
    public boolean equals(Object other)
    {
        if(!(other instanceof Document)) //make sure it's a Document, if not
        {
            return false;
        }
        
        Document rhs = (Document) other;

        return this.name.equals(rhs.name)
            && this.words.equals(rhs.words)
            && this.classes.equals(rhs.classes);
    }

    /**
     * Turns document into a string
     * 
     * @return the buffer as a string
     */
    public String toString()
    {
        StringBuffer buf = new StringBuffer("{");
        buf.append(name);
        buf.append(words.size());
        buf.append(classes);
        buf.append("}");

        return buf.toString();
    }

    /**
     * Reads over the parsed words
     * 
     * @return an iterator for the list of words
     */
    public java.util.Iterator<String> iterator()
    { 
    	return words.iterator();   
    }
}
