package edu.odu.cs.cs350.acmClassifier;

import java.io.File;
import java.util.ArrayList;
import java.util.Scanner;

/**
 * This is supposed to read each document by passing it's path and check the
 * file's formatting.
 *
 *
 * @author Team 7
 *
 */
public class DocumentReader
{
	/**
	 * Taking in the parameter fileName to make sure the fileName has .pdf or .txt
	 * by trying to load the file.
	 *
	 * @param fileName - a single file name
	 *
	 * @try loading the file to see if the fileName does not end in .txt or .pdf
	 * @catch returning an exception and false if the file in not valid
	 * @return true if the file is valid
	 */
	public static boolean isValid(String fileName) //make sure file input is txt or pdf
	{ 
		if(!fileName.endsWith(".txt") && !fileName.endsWith(".pdf")) { return false; } //return false if it doesn't end in correct extension
		
		try { new File(fileName); }           //try loading the file
		catch(Exception ex) { return false; } //if file is not valid, return false
		
		return true; //otherwise, return true
	}

	/**
	 * Will check if all files are valid and end with .pdf or .txt
	 *
	 * @param files - ArrayList of multiple files
	 *
	 * @return if the end of the file is not a .pdf or .txt return false, else
	 * return true
	 */
	public static boolean areValid(String[] files) //make sure all inputted files are valid
	{ 
		
		for(String s : files) //loop through all strings
		{ 
			if(!isValid(s)) { return false; } //if any of them don't end with .txt or .pdf, return false (invalid)
		}
		
		return true; //otherwise, return true (valid)
	}

	/**
	 * Takes in the ArrayList of the raw text of the document and creates an ArrayList
	 * for separting words after spaces.
	 *
	 * @param fileName - name of the file
	 *
	 * @try trying to get the file and see if the file is properly formatted
	 * @catch if the file is not corrcetly formatted the file will return null and
	 * an exception
	 * @return the raw text is return from the requested file/files
	 */
	public static ArrayList<String> giveRawText(String fileName)
	{
		
		ArrayList<String> rawText = new ArrayList<String>(); //initialize return arraylist
		
		Scanner scan;
		try { scan = new Scanner(new File(fileName)); } //get the file
		catch(Exception ex) { return null; }            //if unsuccessful, return null
		
		while(scan.hasNextLine()) //loop until we run out of lines
		{
			String line = scan.nextLine();
			
			String[] words = line.split(" +"); //split into words separated by spaces
			
			for(String s : words) //loop through all the words
			{
				rawText.add(s);     //add each word to the raw text arraylist
			}	
		}
		
		scan.close(); //close the file
		
		return rawText; //return result
	}

	/**
	 * Takes in a name and creates a document to set the raw text in the document and
	 * returns that document.
	 *
	 * @param name - The file name of the document
	 *
	 * @return returns the document withholding the raw text
	 */
	public static Document readDocument(String name)
	{
		Document doc = new Document(name);        //create document
		doc.setWords(giveRawText(doc.getName())); //set words
		return doc;                               //return result
	}
}
