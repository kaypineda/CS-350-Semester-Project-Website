package edu.odu.cs.cs350.acmClassifier;

import java.util.ArrayList;
import java.util.Arrays;

/**
 * Class used for classifying documents through ML
 * @author Team 7
 */

public class Librarian
{
    private ArrayList<Document> docsToClassify;
    private String priorTraining; // file path containing a previous training state, hardcode after training

    /**
     * Default Constructor
     */
    public Librarian()
    {
        docsToClassify = new ArrayList<Document>();
        priorTraining = "";
    }

    /**
     * Constructor given only one document
     * 
     * @param doc Only document supplied
     */
    public Librarian(Document doc)
    {
        docsToClassify = new ArrayList<Document>(Arrays.asList(doc));
        priorTraining = "";
    }

    /**
     * Constructor where only docs to be classified are given
     * 
     * @param docs A document/ list of documents to be classified
     */
    public Librarian(ArrayList<Document> docs)
    {
        docsToClassify = docs;
        priorTraining = "";
    }

    /**
     * Constructor where docts to be classified and a prior training state
     * are given
     * 
     * @param docs A document/ list of documents to be classified
     * @param prior File path of a prior training state
     */
    public Librarian(ArrayList<Document> docs, String prior)
    {
        docsToClassify = docs;
        priorTraining = prior;
    }

    /**
     * Get document(s) to classify
     * 
     * @return Document(s) to classify
     */
    public ArrayList<Document> getDocs()
    {
        return docsToClassify;
    }

    /**
     * Get training state
     * 
     * @return training state
     */
    public String getTrainingState()
    {
        return priorTraining;
    }

    /**
     * Set document(s) to be trained
     * 
     * @param docs Documents to be trained
     */
    public void setDocs(ArrayList<Document> docs)
    {
        docsToClassify = docs;
    }

    /**
     * Set training state to be used
     * 
     * @param training File path to previous training state
     */
    public void setTrainingState(String training)
    {
        priorTraining = training;
    }

    /**
     * Add a document to be classified
     * 
     * @param doc New document to be classified
     */
    public void addDocument(Document doc)
    {
        docsToClassify.add(doc);
    }

    /**
     * Removes a document from documents to be classified
     * 
     * @param doc Document to be removed
     */
    public void removeDocument(Document doc)
    {
        for (int i = 0; i < docsToClassify.size(); i++)
        {
            if (docsToClassify.get(i).equals(doc))
                {
                    docsToClassify.remove(i);
                    return;
                }
        }
    }

    /**
   * Returns hashCode
   * 
   * @return the hashCose of each private member
   */
  public int hashCode()
  {
    return docsToClassify.hashCode();
  }

  /**
   * Trainer clone function
   * 
   * @return a copy of Librarian
   */
  public Librarian clone()
  {
    return new Librarian(docsToClassify, priorTraining);
  }

  /**
   * Check if an object equals trainer
   * 
   * @return true if objects equal/ false if not
   */
  public boolean equals(Object other)
  {
    if(!(other instanceof Librarian)) // if not a librarian
    {
      return false;
    }

    Librarian rhs = (Librarian) other;

    return this.docsToClassify.equals(rhs.docsToClassify)
        && this.priorTraining.equals(rhs.priorTraining);
  }

  /**
     * Turns trainer into a string
     * 
     * @return the buffer as a string
     */
  public String toString()
  {
    StringBuffer buff = new StringBuffer("{");
    buff.append(docsToClassify);
    buff.append(priorTraining);
    buff.append("}");

    return buff.toString();
  }
